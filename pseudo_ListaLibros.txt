

    ListaLibros buscarPorAutor(autor)
        listaAux <- nuevo ListaLibros()
        aux <- primero
        MIENTRAS <aux <> nulo>
            SI <aux.dato.autores = autor>
                listaAux.insertar(aux.clonar)
            FIN SI
            aux <- aux.siguiente
        FIN MIENTRAS
        devuelve listaAux

    eliminarAutor(autor de tipo String)
	aux <- primero
	MIENTRAS <aux <> nulo>
	    SI <aux.dato.autores = autor>
		eliminar(aux.etiqueta)
	    FIN SI
	    aux <- aux.siguiente
	FIN MIENTRAS
